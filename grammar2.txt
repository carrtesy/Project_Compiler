prog -> word "(" ")" block ;
decls -> decls decl
| ;
decl -> vtype words ";" ;
words -> words "," word
 | word ;
vtype -> int | char
| ;
block -> "{" decls slist "}"
| ;
slist -> slist stat
| stat ;
stat -> IF cond THEN block ELSE block
| WHILE cond block
| word "=" expr ";"
| RETURN expr ";"
| ;
cond -> expr ">" expr
| expr "==" expr ;
expr -> term
| term "+" term ;
term -> fact
| fact "*" fact ;
fact -> num
| word
